03 — The Game Loop 3 / 18

Every Tamagonion Script Follows This Pattern

-- 1. Load saved state
load_pet()

-- 2. Ghost-clear (full refresh before first frame)
onion.display_begin(); onion.clear_display(); onion.display_commit()

-- 3. Drain held buttons
local last = onion.buttons()
while last.left or last.right or last.select or last.cancel do
    onion.sleep(50); last = onion.buttons()
end

-- 4. State flags
local tick_acc   = 0
local last_ms    = onion.millis()
local needs_full = true   -- triggers full refresh + clear_display
local needs_bar  = false  -- triggers partial refresh (action bar only)

-- 5. Main loop
while true do
    local now_t = onion.millis()
    local dt    = now_t - last_ms
    if dt < 0 then dt = 0 end
    last_ms = now_t

    -- Accumulate time, fire stat ticks
    tick_acc = tick_acc + dt
    while tick_acc >= TICK_MS do
        tick_acc  = tick_acc - TICK_MS
        tick_stats()
        needs_full = true
    end

    -- Handle buttons (edge detection)
    local btns = onion.buttons()
    if btns.left and not last.left then sel=(sel-1+4)%4; needs_bar=true end
    last = btns

    -- Render exactly once per loop
    if needs_full then
        needs_full = false; needs_bar = false
        render_main(sel, feedback)
    elseif needs_bar then
        needs_bar = false
        render_action_bar_only(sel)
    end

    onion.sleep(20)
end
Key Rule

Never call render inside button handlers. Set a flag, render in one place at the bottom of the loop.

Two-tier refresh

needs_full → clear_display() + full redraw (~1.5 s)
needs_bar → partial redraw of action bar only (~0.5 s)

        cx, cy, r, frame,

        {fill=true, color="black"})

        {fill=true, color="white"})

        onion.display_line(
            cx-7,cy-6, cx-3,cy-2)
        onion.display_line(
            cx-3,cy-6, cx-7,cy-2)
    elseif sleeping then
        onion.display_line(
            cx-7,cy-4, cx-3,cy-4)
    else
        onion.display_circle(
            cx-5,cy-4, 3,
            {fill=true,color="black"})

            cx+5,cy-4, 3,
            {fill=true,color="black"})

        onion.display_line(
            cx-5,cy+6, cx,cy+10)

        onion.display_line(

        cx-(r+2),cy+2, cx-(r+9),cy+dy)

local ASCII_IDLE  = {"  (o_o)  ","  [   ]  ","  /| |\\  "}

local ASCII_SICK  = {"  (x_x)  ","  [   ]  ","   | |   "}
local ASCII_DANCE = {
    {"  \\(^_^)/  ","    |||    ","   /   \\   "},
    {"   (^_^)   ","  \\[   ]/  ","    | |    "},
}

-- Draw the creature
local function draw_ascii(happy, sick)
    local art = sick and ASCII_SICK
             or (happy and ASCII_HAPPY
             or ASCII_IDLE)
    onion.display_lines(art, 88, 40, 16,
        {clear=false, font="small"})
end

-- Animate the dance
local function play_ascii_dance()
    for _ = 1, 4 do
        for _, frame in ipairs(ASCII_DANCE) do
            onion.display_begin()
            onion.display_rect(64,29,136,78,
                {fill=true,color="white"})
            onion.display_lines(frame,72,36,16,
                {clear=false,font="small"})
            onion.display_commit()
            onion.sleep(300)
        end
    end
end

-- ASCII stat bar
local function ascii_bar(val, max, w)
    local f = math.floor(val * w / max)
    return "[" .. string.rep("#",f)
        .. string.rep(".",w-f) .. "]"
end

08 — Stats & Game Logic 8 / 18

Pet State, Ticks & Stage Progression

Pet State

local MAX_STAT = 10
local pet = {
    hunger=MAX_STAT, happiness=MAX_STAT,
    energy=MAX_STAT, health=MAX_STAT,
    age_ticks=0, stage=0,
    alive=true, sleeping=false,
}
local function cstat(v)
    return math.max(0,math.min(MAX_STAT,v))
end

Stage Thresholds

local TICK_MS    = 120000  -- 2 min/tick
local STAGE_TICKS = {
    [0]=0,[1]=3,[2]=20,[3]=60,[4]=120
}
local function stage_for(ticks)
    for i=4,0,-1 do
        if ticks >= STAGE_TICKS[i] then
            return i
        end
    end
    return 0
end

Tick (called every TICK_MS)

local function tick_stats()
    if not pet.alive then return end
    pet.age_ticks = pet.age_ticks + 1
    pet.stage     = stage_for(pet.age_ticks)

    if pet.sleeping then
        pet.energy    = cstat(pet.energy + 3)
        pet.happiness = cstat(pet.happiness + 1)
        if pet.energy >= MAX_STAT then
            pet.sleeping = false
        end
        return
    end
    -- Decay each tick
    pet.hunger = cstat(pet.hunger - 1)
    pet.energy = cstat(pet.energy - 1)
    if pet.age_ticks % 2 == 0 then
        pet.happiness = cstat(pet.happiness-1)
    end
    -- Health consequences
    if pet.hunger    == 0 then
        pet.health = cstat(pet.health - 2) end
    if pet.happiness == 0 then
        pet.health = cstat(pet.health - 1) end
    if pet.hunger >= 7 and
       pet.happiness >= 5 and
       pet.energy   >= 5 then
        pet.health = cstat(pet.health + 1)
    end
    if pet.health == 0 then pet.alive = false end
end

09 — Persistence (NVS) 9 / 18

Save State Across Reboots

save_pet()

local function save_pet()
    if not onion.kv_set then return end
    onion.kv_set("pet", string.format(
        '{"h":%d,"hp":%d,"e":%d,' ..
        '"hl":%d,"a":%d,' ..
        '"alive":%s,"sl":%s}',
        pet.hunger, pet.happiness,
        pet.energy, pet.health,
        pet.age_ticks,
        pet.alive    and "true" or "false",
        pet.sleeping and "true" or "false"))
end

NVS Rules

  • Key length ≤ 15 characters
  • Value is any string
  • kv_get(key) → string or nil
  • kv_del(key) → delete

load_pet()

local function load_pet()
    if not onion.kv_get then return end
    local s = onion.kv_get("pet")
    if not s or s == "" then return end
    local function n(k)
        return tonumber(
            s:match('"'..k..'":(%d+)'))
    end
    if n("h")  then pet.hunger    = cstat(n("h"))  end
    if n("hp") then pet.happiness = cstat(n("hp")) end
    if n("e")  then pet.energy    = cstat(n("e"))  end
    if n("hl") then pet.health    = cstat(n("hl")) end
    if n("a")  then pet.age_ticks = n("a")         end
    pet.alive    = (s:find('"alive":true')  ~= nil)
    pet.sleeping = (s:find('"sl":true') ~= nil)
    pet.stage    = stage_for(pet.age_ticks)
end
When to save

Call save_pet() after every SELECT action and before onion.release_display(). NVS writes survive power cycles.

10 — Input Handling 10 / 18

Edge Detection & Button Actions

local sel  = 0       -- 0=FEED  1=PLAY  2=SLEEP  3=MED
local last = onion.buttons()

-- ── Inside the main loop: ────────────────────────────────────
local btns = onion.buttons()

-- Edge detection: fires ONCE when button goes down
if btns.left  and not last.left  then sel = (sel-1+4)%4; needs_bar = true end
if btns.right and not last.right then sel = (sel+1)%4;   needs_bar = true end
if btns.up    and not last.up    then sel = 0;            needs_bar = true end
if btns.down  and not last.down  then sel = 2;            needs_bar = true end

if btns.select and not last.select then
    local msg = perform_action(sel)
    set_feedback(msg)
    save_pet()
    needs_full = true
end

if btns.cancel and not last.cancel then
    save_pet()
    onion.release_display()  -- return display to Onion OS
    return                   -- exit script
end

last = btns   -- update edge-detection state
Button names

left   right   up   down   select   cancel

Edge detection

btns.x and not last.x fires only on the frame the button transitions from released → pressed. Without it, holding a button fires every 20 ms.

    onion.sound_play_tone(523, 100, 0.5)
    onion.sound_play_tone(659, 100, 0.5)
    onion.sound_play_tone(784, 200, 0.7) -- G5
